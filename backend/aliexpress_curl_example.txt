# Example curl command for AliExpress API via RapidAPI
# Replace YOUR_RAPIDAPI_KEY with your actual RapidAPI key

# Search for "iphone 16 pro max" products
curl --request GET \
  --url 'https://aliexpress-datahub.p.rapidapi.com/products/search?keywords=iphone%2016%20pro%20max&page=1&sort=best_match' \
  --header 'X-RapidAPI-Key: YOUR_RAPIDAPI_KEY' \
  --header 'X-RapidAPI-Host: aliexpress-datahub.p.rapidapi.com'

# The same request can be used in Python with the requests library:
'''python
